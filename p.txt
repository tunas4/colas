def insertar_cola_doble(self, v, l):
        # 0 = fin
        # 1 = inicio
        if self.inicio == -1:
            self.inicio = self.fin = 0
            self.cola[self.inicio] = v
        else:
            if l == 0:
                if self.fin < self.max - 1:
                    self.fin += 1
                    self.cola[self.fin] = v
                else:
                    return "Cola llena por el final"
            else:
                if self.inicio > 0:
                    self.inicio -= 1
                    self.cola[self.inicio] = v
                else:
                    return "Cola llena por el principio"
                
    def insertar_cola_circular(self, v):
        if (self.inicio == 0 and self.fin == self.max - 1) or (self.fin == self.inicio - 1):
            return "Cola llena"
        elif self.inicio == self.max - 1:
            self.fin = 0
        else:
            self.inicio += 1
            
        if self.inicio == -1:
            self.inicio = self.inicio + 1

        self.cola[self.inicio] = v

def eliminar_cola_doble(self, l):
        if self.inicio > -1:
            if self.inicio == self.fin:
                t = self.cola[self.inicio]
                self.inicio = self.fin = -1
            else:
                if l == 0:
                    t = self.cola[self.fin]
                    self.fin -= 1
                else:
                    t = self.cola[self.inicio]
                    self.inicio += 1
            return t
        else:
            return "Cola vacia"
        
    def eliminar_cola_circular(self):
        if self.fin > -1:
            t = self.cola[self.inicio]
            if self.inicio == self.fin:
                self.inicio = self.fin = -1
            elif self.inicio == self.max - 1:
                self.inicio = 0
            else:
                self.inicio += 1
            return t
        else:
            t = None
        return t
